Provide an Iface.

A naked instance pointer is allocated and wrapped in a shared_ptr. It is also given a custom deleter, to dispose of the naked pointer with dispose(Iface *).

Implements sauce::internal::InjectionBinding< Dependency, Scope >.

{
Deleter deleter(sauce::static_pointer_cast<New>(binding));
Impl * impl = allocator.allocate(1);
Ptr provided(new(impl) Impl(), deleter);
return provided;
}
